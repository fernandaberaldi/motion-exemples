JavaScript · Counter 
// easing function
function easeOutExpo(t) {
  return t >= 1 ? 1 : 1 - Math.pow(2, -10 * t);
}

// formata número (vírgula BR)
function fmt(v, dec) {
  return dec
    ? v.toFixed(1).replace('.', ',')
    : Math.floor(v) + '';
}

// anima um contador
function animateCounter(el, target, dec, delay) {
  const DUR = 1900; // ms
  let t0 = null;
  setTimeout(() => {
    (function tick(ts) {
      if (!t0) t0 = ts;
      const p = Math.min((ts - t0) / DUR, 1);
      el.textContent = fmt(easeOutExpo(p) * target, dec);
      if (p < 1) requestAnimationFrame(tick);
      else el.textContent = fmt(target, dec);
    })(performance.now());
  }, delay);
}

// dispara ao entrar na viewport
let played = false;
const io = new IntersectionObserver(([e]) => {
  if (e.isIntersecting && !played) {
    played = true;
    counters.forEach((cfg, i) =>
      animateCounter(

        cfg.target, cfg.decimal, i * 130
      )
    );
  }
}, { threshold: 0.3 });

JavaScript · Tilt 3D 
const MAX_TILT = 6; // graus máx

document.querySelectorAll('.card').forEach(card => {

  card.addEventListener('mousemove', e => {
    const r  = card.getBoundingClientRect();
    const dx = (e.clientX - r.left - r.width  / 2) / (r.width  / 2);
    const dy = (e.clientY - r.top  - r.height / 2) / (r.height / 2);

    const rotX = (-dy * MAX_TILT).toFixed(2);
    const rotY = (  dx * MAX_TILT).toFixed(2);

    card.style.transform =
      `perspective(900px) rotateX(${rotX}deg)
       rotateY(${rotY}deg) scale3d(1.02,1.02,1.02)`;
    card.style.transition =
      'transform .08s ease, box-shadow .15s ease';
    card.style.boxShadow =
      `${-dx*6}px ${-dy*6+12}px 32px rgba(113,2,82,.16),
       ${-dx*2}px ${-dy*2+4}px 10px rgba(113,2,82,.08)`;

    // shimmer segue cursor
    card.style.setProperty('--mx',
      ((e.clientX - r.left) / r.width  * 100).toFixed(1) + '%');
    card.style.setProperty('--my',
      ((e.clientY - r.top)  / r.height * 100).toFixed(1) + '%');
  });

  card.addEventListener('mouseleave', () => {
    card.style.transition =
      'transform .55s cubic-bezier(.03,.98,.52,.99),
       box-shadow .4s ease';
    card.style.transform =
      'perspective(900px) rotateX(0deg)
       rotateY(0deg) scale3d(1,1,1)';
    card.style.boxShadow = '';
  });
});

/* Wrapper — define perspectiva */
.card-wrap {
  perspective: 900px;
  min-width: 0; /* evita overflow em grid/flex */
}

/* Card base */
.card {
  border-radius: 20px;
  transform-style: preserve-3d;
  will-change: transform;
  transition: transform .45s cubic-bezier(.03,.98,.52,.99),
              box-shadow .3s ease;
  overflow: hidden;
  cursor: pointer;
}

/* Shimmer de luz segue o cursor */
.card::before {
  content: '';
  position: absolute; inset: 0;
  border-radius: inherit;
  background: radial-gradient(
    circle at var(--mx, 50%) var(--my, 50%),
    rgba(255,255,255,.22) 0%, transparent 60%
  );
  opacity: 0;
  transition: opacity .2s ease;
  pointer-events: none;
  z-index: 10;
}
.card:hover::before { opacity: 1; }

Checklist de implementação

  • ✓ Wrapper .card-wrap com perspective: 900px — sem isso o tilt não funciona em 3D
  • ✓ CSS custom properties --mx e --my setadas via JS no mousemove para o shimmer
  • ✓ Counter só dispara 1× via flag played — evita replay indesejado ao scrollar para cima
  • ✓ Decimais formatados com vírgula BR via .replace('.', ',')
  • ✓ Grid grid-template-columns: 280px 240px 160px 240px — col 3 mais estreita é intencional
  • ⚠ overflow: hidden no .card corta a sombra no tilt — a sombra é aplicada via box-shadow inline (não é cortada)
  • ⚠ Tilt desabilitado em touch — adicionar detecção de ontouchstart e pular listeners se mobile
  • ⚠ Respeitar prefers-reduced-motion — envolver o JS de tilt e counter em checagem da media query
