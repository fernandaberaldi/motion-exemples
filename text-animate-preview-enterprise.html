3 · HTML Embed no Webflow

Adicione um bloco Embed e cole o código abaixo. Ele gera os spans automaticamente com os pesos corretos:

<h1 id="blur-text" class="blur-heading"></h1>

<script>
  // Segmentos: [texto, peso]  ("semi" = 600, "bold" = 700, "" = 400)
  const segments = [
    ["Grandes empresas ", "semi"],
    ["tratam a saúde mental de ", ""],
    ["forma estratégica.", "bold"]
  ];

  const el = document.getElementById("blur-text");
  let delay = 0;

  segments.forEach(([text, weight]) => {
    text.split(" ").forEach(word => {
      if (!word) return;
      const span = document.createElement("span");
      span.className = "blur-word" + (weight ? " " + weight : "");
      span.textContent = word + " ";
      span.style.animationDelay = delay + "ms";
      el.appendChild(span);
      delay += 80;
    });
  });
</script>

@keyframes blurIn {
  to {
    opacity: 1;
    filter: blur(0px);
    transform: translateY(0px);
  }

6 · Opcional — Disparar ao entrar na viewport

Substitua o script do passo 3 por este para animar ao rolar a página:

<script>
  const segments = [
    ["Grandes empresas ", "semi"],
    ["tratam a saúde mental de ", ""],
    ["forma estratégica.", "bold"]
  ];

  const el = document.getElementById("blur-text");
  let delay = 0;

  segments.forEach(([text, weight]) => {
    text.split(" ").forEach(word => {
      if (!word) return;
      const span = document.createElement("span");
      span.className = "blur-word" + (weight ? " " + weight : "");
      span.textContent = word + " ";
      span.style.animationDelay = delay + "ms";
      span.style.animationPlayState = "paused";
      el.appendChild(span);
      delay += 80;
    });
  });

  new IntersectionObserver(([entry]) => {
    if (entry.isIntersecting) {
      el.querySelectorAll(".blur-word").forEach(s => {
        s.style.animationPlayState = "running";
      });
    }
  }, { threshold: 0.2 }).observe(el);
</script>
